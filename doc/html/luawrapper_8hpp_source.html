7 // API Summary:
8 //
9 // LuaWrapper is a library designed to help bridge the gab between Lua and
10 // C++. It is designed to be small (a single header file), simple, fast,
11 // and typesafe. It has no external dependencies, and does not need to be
12 // precompiled; the header can simply be dropped into a project and used
13 // immediately. It even supports class inheritance to a certain degree. Objects
14 // can be created in either Lua or C++, and passed back and forth.
15 //
16 // The main functions of interest are the following:
17 // luaW_is<T>
18 // luaW_to<T>
19 // luaW_check<T>
20 // luaW_push<T>
21 // luaW_register<T>
22 // luaW_setfuncs<T>
23 // luaW_extend<T, U>
24 // luaW_hold<T>
25 // luaW_release<T>
26 //
27 // These functions allow you to manipulate arbitrary classes just like you
28 // would the primitive types (e.g. numbers or strings). If you are familiar
29 // with the normal Lua API the behavior of these functions should be very
30 // intuative.
31 //
32 // For more information see the README and the comments below

94 // typid to compare types, I use the clever trick of using the cast to compare

152 // Analogous to lua_is(boolean|string|*)
153 //
154 // Returns 1 if the value at the given acceptable index is of type T (or if
155 // strict is false, convertable to type T) and 0 otherwise.
156 template <typename T>
157 bool luaW_is(lua_State *L, int index, bool strict = false)
158 {
159  bool equal = false;// lua_isnil(L, index);
160  if (!equal && lua_isuserdata(L, index) && lua_getmetatable(L, index))
161  {
162  // ... ud ... udmt
163  luaL_getmetatable(L, LuaWrapper<T>::classname); // ... ud ... udmt Tmt
164  equal = lua_rawequal(L, -1, -2) != 0;
165  if (!equal && !strict)
166  {
167  lua_getfield(L, -2, LUAW_EXTENDS_KEY); // ... ud ... udmt Tmt udmt.extends
168  for (lua_pushnil(L); lua_next(L, -2); lua_pop(L, 1))
169  {
170  // ... ud ... udmt Tmt udmt.extends k v
171  equal = lua_rawequal(L, -1, -4) != 0;
172  if (equal)
173  {
174  lua_pop(L, 2); // ... ud ... udmt Tmt udmt.extends
175  break;
176  }
177  }
178  lua_pop(L, 1); // ... ud ... udmt Tmt
179  }
180  lua_pop(L, 2); // ... ud ...
181  }
182  return equal;
183 }

185 // Analogous to lua_to(boolean|string|*)
186 //
187 // Converts the given acceptable index to a T*. That value must be of (or
188 // convertable to) type T; otherwise, returns NULL.
189 template <typename T>
190 T* luaW_to(lua_State* L, int index, bool strict = false)
191 {
192  if (luaW_is<T>(L, index, strict))
193  {
194  luaW_Userdata* pud = static_cast<luaW_Userdata*>(lua_touserdata(L, index));
195  luaW_Userdata ud;
196  while (!strict && LuaWrapper<T>::cast != pud->cast)
197  {
198  ud = pud->cast(*pud);
199  pud = &ud;
200  }
201  return static_cast<T*>(pud->data);
202  }
203  return NULL;
204 }

207 //

246 // Analogous to lua_push(boolean|string|*)
247 //
248 // Pushes a userdata of type T onto the stack. If this object already exists in
249 // the Lua environment, it will assign the existing storage table to it.
250 // Otherwise, a new storage table will be created for it.
251 template <typename T>
252 void luaW_push(lua_State* L, T* obj)
253 {
254  if (obj)
255  {
256  LuaWrapper<T>::identifier(L, obj); // ... id
257  luaW_wrapperfield<T>(L, LUAW_CACHE_KEY); // ... id cache
258  lua_pushvalue(L, -2); // ... id cache id
259  lua_gettable(L, -2); // ... id cache obj
260  if (lua_isnil(L, -1))
261  {
262  // Create the new luaW_userdata and place it in the cache
263  lua_pop(L, 1); // ... id cache
264  lua_insert(L, -2); // ... cache id
265  luaW_Userdata* ud = static_cast<luaW_Userdata*>(lua_newuserdata(L, sizeof(luaW_Userdata))); // ... cache id obj
266  ud->data = obj;
267  ud->cast = LuaWrapper<T>::cast;
268  lua_pushvalue(L, -1); // ... cache id obj obj
269  lua_insert(L, -4); // ... obj cache id obj
270  lua_settable(L, -3); // ... obj cache
271 
272  luaL_getmetatable(L, LuaWrapper<T>::classname); // ... obj cache mt
273  lua_setmetatable(L, -3); // ... obj cache
274 
275  lua_pop(L, 1); // ... obj
276  }
277  else
278  {
279  lua_replace(L, -3); // ... obj cache
280  lua_pop(L, 1); // ... obj
281  }
282  }
283  else
284  {
285  lua_pushnil(L);
286  }
287 }

289 // Instructs LuaWrapper that it owns the userdata, and can manage its memory.
290 // When all references to the object are removed, Lua is free to garbage
291 // collect it and delete the object.
292 //
293 // Returns true if luaW_hold took hold of the object, and false if it was
294 // already held
295 template <typename T>
296 bool luaW_hold(lua_State* L, T* obj)
297 {
298  luaW_wrapperfield<T>(L, LUAW_HOLDS_KEY); // ... holds
299  LuaWrapper<T>::identifier(L, obj); // ... holds id
300  lua_pushvalue(L, -1); // ... holds id id
301  lua_gettable(L, -3); // ... holds id hold
302  // If it's not held, hold it
303  if (!lua_toboolean(L, -1))
304  {
305  // Apply hold boolean
306  lua_pop(L, 1); // ... holds id
307  lua_pushboolean(L, true); // ... holds id true
308  lua_settable(L, -3); // ... holds
309  lua_pop(L, 1); // ...
310  return true;
311  }
312  lua_pop(L, 3); // ...
313  return false;
314 }

316 // Releases LuaWrapper's hold on an object. This allows the user to remove
317 // all references to an object in Lua and ensure that Lua will not attempt to
318 // garbage collect it.
319 //
320 // This function takes the index of the identifier for an object rather than
321 // the object itself. This is because needs to be able to run after the object
322 // has already been deallocated. A wrapper is provided for when it is more
323 // convenient to pass in the object directly.
324 template <typename T>
325 void luaW_release(lua_State* L, int index)
326 {
327  luaW_wrapperfield<T>(L, LUAW_HOLDS_KEY); // ... id ... holds
328  lua_pushvalue(L, luaW_correctindex(L, index, 1)); // ... id ... holds id
329  lua_pushnil(L); // ... id ... holds id nil
330  lua_settable(L, -3); // ... id ... holds
331  lua_pop(L, 1); // ... id ...
332 }

405 // This function is called from Lua, not C++
406 //
407 // The default metamethod to call when indexing into lua userdata representing
408 // an object of type T. This will first check the userdata's environment table
409 // and if it's not found there it will check the metatable. This is done so
410 // individual userdata can be treated as a table, and can hold thier own
411 // values.
412 template <typename T>
413 int luaW_index(lua_State* L)
414 {
415  // obj key
416  T* obj = luaW_to<T>(L, 1);
417  luaW_wrapperfield<T>(L, LUAW_STORAGE_KEY); // obj key storage
418  LuaWrapper<T>::identifier(L, obj); // obj key storage id
419  lua_gettable(L, -2); // obj key storage store
420 
421  // Check if storage table exists
422  if (!lua_isnil(L, -1))
423  {
424  lua_pushvalue(L, -3); // obj key storage store key
425  lua_gettable(L, -2); // obj key storage store store[k]
426  }
427 
428  // If either there is no storage table or the key wasn't found
429  // then fall back to the metatable
430  if (lua_isnil(L, -1))
431  {
432  lua_settop(L, 2); // obj key
433  lua_getmetatable(L, -2); // obj key mt
434  lua_pushvalue(L, -2); // obj key mt k
435  lua_gettable(L, -2); // obj key mt mt[k]
436  }
437  return 1;
438 }

558 // Run luaW_register or luaW_setfuncs to create a table and metatable for your
559 // class. These functions create a table with filled with the function from
560 // the table argument in addition to the functions new and build (This is
561 // generally for things you think of as static methods in C++). The given
562 // metatable argument becomes a metatable for each object of your class. These
563 // can be thought of as member functions or methods.
564 //
565 // You may also supply constructors and destructors for classes that do not
566 // have a default constructor or that require special set up or tear down. You
567 // may specify NULL as the constructor, which means that you will not be able
568 // to call the new function on your class table. You will need to manually push
569 // objects from C++. By default, the default constructor is used to create
570 // objects and a simple call to delete is used to destroy them.
571 //
572 // By default LuaWrapper uses the address of C++ object to identify unique
573 // objects. In some cases this is not desired, such as in the case of
574 // shared_ptrs. Two shared_ptrs may themselves have unique locations in memory
575 // but still represent the same object. For cases like that, you may specify an
576 // identifier function which is responsible for pushing a key representing your
577 // object on to the stack.
578 //
579 // luaW_register will set table as the new value of the global of the given
580 // name. luaW_setfuncs is identical to luaW_register, but it does not set the
581 // table globally. As with luaL_register and luaL_setfuncs, both funcstions
582 // leave the new table on the top of the stack.
583 template <typename T>

647 // luaW_extend is used to declare that class T inherits from class U. All
648 // functions in the base class will be available to the derived class (except
649 // when they share a function name, in which case the derived class's function
650 // wins). This also allows luaW_to<T> to cast your object apropriately, as
651 // casts straight through a void pointer do not work.
652 template <typename T, typename U>
653 void luaW_extend(lua_State* L)
654 {
655  if(!LuaWrapper<T>::classname)
656  luaL_error(L, "attempting to call extend on a type that has not been registered");
657 
658  if(!LuaWrapper<U>::classname)
659  luaL_error(L, "attempting to extend %s by a type that has not been registered", LuaWrapper<T>::classname);
660 
661  LuaWrapper<T>::cast = luaW_cast<T, U>;
662  LuaWrapper<T>::identifier = luaW_identify<T, U>;
663  LuaWrapper<T>::postconstructorrecurse = luaW_postconstructorinternal<U>;
664 
665  luaL_getmetatable(L, LuaWrapper<T>::classname); // mt
666  luaL_getmetatable(L, LuaWrapper<U>::classname); // mt emt
667 
668  // Point T's metatable __index at U's metatable for inheritance
669  lua_newtable(L); // mt emt {}
670  lua_pushvalue(L, -2); // mt emt {} emt
671  lua_setfield(L, -2, "__index"); // mt emt {}
672  lua_setmetatable(L, -3); // mt emt
673 
674  // Set up per-type tables to point at parent type
675  lua_getfield(L, LUA_REGISTRYINDEX, LUAW_WRAPPER_KEY); // ... LuaWrapper
676 
677  lua_getfield(L, -1, LUAW_STORAGE_KEY); // ... LuaWrapper LuaWrapper.storage
678  lua_getfield(L, -1, LuaWrapper<U>::classname); // ... LuaWrapper LuaWrapper.storage U
679  lua_setfield(L, -2, LuaWrapper<T>::classname); // ... LuaWrapper LuaWrapper.storage
680  lua_pop(L, 1); // ... LuaWrapper
681 
682  lua_getfield(L, -1, LUAW_HOLDS_KEY); // ... LuaWrapper LuaWrapper.holds
683  lua_getfield(L, -1, LuaWrapper<U>::classname); // ... LuaWrapper LuaWrapper.holds U
684  lua_setfield(L, -2, LuaWrapper<T>::classname); // ... LuaWrapper LuaWrapper.holds
685  lua_pop(L, 1); // ... LuaWrapper
686 
687  lua_getfield(L, -1, LUAW_CACHE_KEY); // ... LuaWrapper LuaWrapper.cache
688  lua_getfield(L, -1, LuaWrapper<U>::classname); // ... LuaWrapper LuaWrapper.cache U
689  lua_setfield(L, -2, LuaWrapper<T>::classname); // ... LuaWrapper LuaWrapper.cache
690 
691  lua_pop(L, 2); // ...
692 
693  // Make a list of all types that inherit from U, for type checking
694  lua_getfield(L, -2, LUAW_EXTENDS_KEY); // mt emt mt.extends
695  lua_pushvalue(L, -2); // mt emt mt.extends emt
696  lua_setfield(L, -2, LuaWrapper<U>::classname); // mt emt mt.extends
697  lua_getfield(L, -2, LUAW_EXTENDS_KEY); // mt emt mt.extends emt.extends
698  for (lua_pushnil(L); lua_next(L, -2); lua_pop(L, 1))
699  {
700  // mt emt mt.extends emt.extends k v
701  lua_pushvalue(L, -2); // mt emt mt.extends emt.extends k v k
702  lua_pushvalue(L, -2); // mt emt mt.extends emt.extends k v k
703  lua_rawset(L, -6); // mt emt mt.extends emt.extends k v
704  }
705 
706  lua_pop(L, 4); // mt emt
707 }
