214 //
215 // A set of trivial getter and setter templates. These templates are designed
216 // to call trivial getters or setters.
217 //
218 // There are four forms:
219 // 1. Getting or setting a public member variable that is of a primitive type
220 // 2. Getting or setting a public member variable that is a pointer to an
221 // object
222 // 3. Getting or setting a private member variable that is of a primitive type
223 // through a getter or setter
224 // 3. Getting or setting a private member variable that is is a pointer to an
225 // object through a getter or setter
226 //
227 // The interface to all of them is the same however. In addition to plain
228 // getter and setter functions, there is a getset which does both - if an
229 // argument is supplied it attempts to set the value, and in either case it
230 // returns the value. In your lua table declaration in C++ rather than write
231 // individiual wrappers for each getter and setter you may do the following:
232 //
233 // static luaL_reg Foo_metatable[] =
234 // {
235 // { "GetBar", luaU_get<Foo, bool, &Widget::GetBar> },
236 // { "SetBar", luaU_set<Foo, bool, &Widget::SetBar> },
237 // { "Bar", luaU_getset<Foo, bool, &Widget::GetBar, &Widget::SetBar> },
238 // { NULL, NULL }
239 // }
240 //
241 // Getters and setters must have one of the following signatures:
242 // void T::Setter(U value);
243 // void T::Setter(U* value);
244 // void T::Setter(const U& value);
245 // U Getter() const;
246 // U* Getter() const;
247 //
248 // In this example, the variable Foo::bar is private and so it is accessed
249 // through getter and setter functions. If Foo::bar were public, it could be
250 // accessed directly, like so:
251 //
252 // static luaL_reg Foo_metatable[] =
253 // {
254 // { "GetBar", luaU_get<Foo, bool, &Widget::bar> },
255 // { "SetBar", luaU_set<Foo, bool, &Widget::bar> },
256 // { "Bar", luaU_getset<Foo, bool, &Widget::bar> },
257 // }
258 //
259 // In a Lua script, you can now use foo:GetBar(), foo:SetBar() and foo:Bar()
260 //

532 //
533 // luaU_func is a special macro that expands into a simple function wrapper.
534 // Unlike the getter setters above, you merely need to name the function you
535 // would like to wrap.
536 //
537 // For example,
538 //
539 // struct Foo
540 // {
541 // int DoSomething(int, const char*);
542 // };
543 //
544 // static luaL_reg Foo_metatable[] =
545 // {
546 // { "DoSomething", luaU_func(&Foo::DoSomething) },
547 // { NULL, NULL }
548 // }
549 //
550 // void RegisterFoo(lua_State* L)
551 // {
552 // luaW_register<Foo>(L, "Foo", NULL, Foo_metatable);
553 // }
554 //
555 // This macro will expand based on the function signature of Foo::DoSomething
556 // In this example, it would expand into the following wrapper:
557 //
558 // luaU_push(luaW_check<T>(L, 1)->DoSomething(luaU_check<int>(L, 2), luaU_check<const char*>(L, 3)));
559 // return 1;
560 //
561 // In this example there is only one member function called DoSomething. In some
562 // cases there may be multiple overrides for a function. For those cases, an
563 // additional macro has been provided, luaU_funcsig, which takes the entire
564 // function signature. The arguments to the macro reflect the order they would
565 // appear in the function signature: return type, type name, function name, and
566 // finally a list of all the argument types. For example:
567 //
568 // struct Foo
569 // {
570 // int DoSomething (const char*);
571 // int DoSomething (const char*, int);
572 // };
573 //
574 // const struct luaL_Reg Foo_metatable[] =
575 // {
576 // {"DoSomething1", luaU_funcsig(int, Foo, DoSomething, const char*)) },
577 // {"DoSomething1", luaU_funcsig(int, Foo, DoSomething, const char*, int)) },
578 // { NULL, NULL }
579 // };
580 //
581 // There`s also support for static and freestanding functions. Macros luaU_staticfunc
582 // and luaU_staticfuncsig work equally to luaU_func and luaU_funcsig, except for that
583 // you need to provide a separate metatable for static functions.
584 // For example,
585 //
586 // struct Foo
587 // {
588 // int DoSomething(int, const char*);
589 // static void DoSomethingElse(int a, int b, float c);
590 // };
591 //
592 // static luaL_reg Foo_metatable[] =
593 // {
594 // { "DoSomething", luaU_func(&Foo::DoSomething) },
595 // { NULL, NULL }
596 // };
597 //
598 // static luaL_reg Foo_metatable_static[] =
599 // {
600 // { "DoSomethingElse", luaU_staticfunc(&Foo::DoSomethingElse) },
601 // { NULL, NULL }
602 // };
603 //
604 // void RegisterFoo(lua_State* L)
605 // {
606 // luaW_register<Foo>(L, "Foo", Foo_metatable_static, Foo_metatable);
607 // }
608 //
609 // After that you will be able to use Foo class from Lua like this:
610 // local foo = Foo.new()
611 // foo:DoSomething(42, 'The Ultimate Question of Life, the Universe, and Everything.') -- member function call
612 // Foo:DoSomethingElse(30, 12, 3.1459) -- Static function call
613 //
614 // These macros and it's underlying templates are somewhat experimental and some
615 // refinements are probably needed. There are cases where it does not
616 // currently work and I expect some changes can be made to refine its behavior.
617 //
618

718 //
719 // Calls the copy constructor for an object of type T.
720 // Arguments may be passed in, in case they're needed for the postconstructor
721 //
722 // e.g.
723 //
724 // C++:
725 // luaL_reg Foo_Metatable[] =
726 // {
727 // { "clone", luaU_clone<Foo> },
728 // { NULL, NULL }
729 // };
730 //
731 // Lua:
732 // foo = Foo.new()
733 // foo2 = foo:clone()
734 //
735 template <typename T>
736 int luaU_clone(lua_State* L)
737 {
738  // obj ...
739  T* obj = new T(*luaW_check<T>(L, 1));
740  lua_remove(L, 1); // ...
741  int numargs = lua_gettop(L);
742  luaW_push<T>(L, obj); // ... clone
743  luaW_hold<T>(L, obj);
744  luaW_postconstructor<T>(L, numargs);
745  return 1;
746 }

749 //
750 // luaU_build is intended to be used to initialize many values by passing in a
751 // table. They keys of the table are used as function names, and values are
752 // used as arguments to the function. This is intended to be used on functions
753 // that are simple setters.
754 //
755 // For example, if luaU_build is set as the post constructor, you can
756 // initialize an object as so:
757 //
758 // f = Foo.new
759 // {
760 // X = 10;
761 // Y = 20;
762 // }
763 //
764 // After the object is constructed, luaU_build will do the equivalent of
765 // calling f:X(10) and f:Y(20).
766 //
767 template <typename T>
768 int luaU_build(lua_State* L)
769 {
770  // obj {}
771  lua_insert(L, -2); // {} obj
772  if (lua_type(L, 1) == LUA_TTABLE)
773  {
774  for (lua_pushnil(L); lua_next(L, 1); lua_pop(L, 1))
775  {
776  // {} obj k v
777  lua_pushvalue(L, -2); // {} obj k v k
778  lua_gettable(L, -4); // {} obj k v ud[k]
779  lua_pushvalue(L, -4); // {} obj k v ud[k] ud
780  lua_pushvalue(L, -3); // {} obj k v ud[k] ud v
781  lua_call(L, 2, 0); // {} obj k v
782  }
783  // {} ud
784  }
785  return 0;
786 }

789 //
790 // Takes the object of type T at the top of the stack and stores it in on a
791 // table with the name storagetable, on the table at the specified index.
792 //
793 // You may manually call luaW_hold and luaW_release to handle pointer
794 // ownership, but it is often easier to simply store a Lua userdata on a table
795 // that is owned by its parent. This ensures that your object will not be
796 // prematurely freed, and that it can only be destoryed after its parent.
797 //
798 // e.g.
799 //
800 // Foo* parent = luaW_check<Foo>(L, 1);
801 // Bar* child = luaW_check<Bar>(L, 2);
802 // if (parent && child)
803 // {
804 // luaU_store<Bar>(L, 1, "Children");
805 // parent->AddChild(child);
806 // }
807 //
808 template <typename T>
809 void luaU_store(lua_State* L, int index, const char* storagetable, const char* key = NULL)
810 {
811  // ... store ... obj
812  lua_getfield(L, index, storagetable); // ... store ... obj store.storagetable
813  if (key)
814  lua_pushstring(L, key); // ... store ... obj store.storagetable key
815  else
816  LuaWrapper<T>::identifier(L, luaW_to<T>(L, -2)); // ... store ... obj store.storagetable key
817  lua_pushvalue(L, -3); // ... store ... obj store.storagetable key obj
818  lua_settable(L, -3); // ... store ... obj store.storagetable
819  lua_pop(L, 1); // ... store ... obj
820 }
